function DidItChange(Checker, Player)
	pcall(function()
		workspace.Characters:WaitForChild(Player.Name, 15)
		Player:WaitForChild("Information"):WaitForChild("Tag", 15)
		local CharRacter = workspace.Characters:WaitForChild(Player.Name, 15)
		CharRacter:WaitForChild("Head", 15)
		CheckIfStreak(CharRacter, Player)

		if Player ~= game:GetService("Players").LocalPlayer then
			Player.CharacterAdded:Connect(function(Char)
				Char:WaitForChild("I_LOADED_I")
				CheckIfStreak(Char, Player)
			end)
		end

		Player.Information.Tag.Changed:Connect(function()
			wait()
			_G[Player.UserId].Animating = false
			local Character = Player.Character
			if Character.Head:FindFirstChild("Streak-Kill") and Character.Head:FindFirstChild("Streak-Kill").TypeTag.Value ~= Player.Information.Tag.Value and Player.Information.Streaks.Value > 0 then
				if Character.Head:FindFirstChild("Streak-Kill") then
					Character.Head:FindFirstChild("Streak-Kill"):Destroy()
				end
				local GUI = RPL.TagText[Player.Information.Tag.Value]:Clone()
				GUI.Parent = Character.Head
				GUI.Name = "Streak-Kill"
				if game.Players[Character.Name].Information.Streaks.Value <= 1 then
					GUI.Text.Text = "Kill Streak - "..Player.Information.Streaks.Value
				else
					GUI.Text.Text = "Kill Streaks - "..Player.Information.Streaks.Value
				end
			end
		end)


		Player.Information.Streaks.Changed:Connect(function()
			wait()

			local Character = Player.Character
			if Player:FindFirstChild("Admn") == nil and Player:FindFirstChild("Moderator") == nil then

				if Player.Information.Streaks.Value == 0 then
					if Character.Head:FindFirstChild("Streak-Kill") then
						Character.Head:FindFirstChild("Streak-Kill"):Destroy()
					end
				else

					ReactiveTags(Player)
					if Character.Head:FindFirstChild("Streak-Kill") == nil then
						local GUI = RPL.TagText[Player.Information.Tag.Value]:Clone()
						GUI.Parent = Character.Head
						GUI.Name = "Streak-Kill"
						GUI.Text.Text = "Kill Streak - "..Player.Information.Streaks.Value

		end)

	end)

local TradeUI = script.Parent:WaitForChild("TradeUI")

function AddSoundToClick()
	TradeUI.Sound:Play()
end

function RemoveTheTrade(button, val, WhichC, FL, origin)
	local YoungOne
	YoungOne = button.MouseButton1Click:Connect(function()
		if cooldown == false then
		game.ReplicatedStorage.MainRemote:FireServer("RemoveTrade", FL, val, WhichC)
		AddSoundToClick()
		if origin:FindFirstChild("Raycodex") then
			origin["Raycodex"]:Destroy()
		end

			YoungOne:Disconnect()
		end
	end)
end

function AddUpToTrade(button, val, WhichC, FL)
	local YoungOne
	YoungOne = button.MouseButton1Click:Connect(function()
		if cooldown == false then
		game.ReplicatedStorage.MainRemote:FireServer("AddUpTrade", FL, val, WhichC)
		AddSoundToClick()
		local sff = Instance.new("StringValue", button)
		sff.Name = "Raycodex"
			YoungOne:Disconnect()
		end
	end)

	TradeUI.Changed:Connect(function()
		if TradeUI.Visible == false then
			CurrentThing = 0
			YoungOne:Disconnect()
		end
	end)
end

function AddButton(button, FL, NAME)
	local YoungOne
	YoungOne = button.MouseButton1Click:Connect(function()
		if cooldown == false then
			game.ReplicatedStorage.MainRemote:FireServer(NAME, FL)
		end
		AddSoundToClick()
		
	end)

	TradeUI.Changed:Connect(function()
		if TradeUI.Visible == false then
			YoungOne:Disconnect()
		end
	end)
end

function LoadTradesUp(OtherPerson, WhichC, Folder)
	if game.Players:FindFirstChild(OtherPerson) then
		TradeUI.Visible = true
		for i, v in pairs(TradeUI.SPL.Inventory:GetChildren()) do
			if v:IsA("Frame") then
				v:Destroy()
			end
		end
		for i, v in pairs(TradeUI.FPL.Inventory:GetChildren()) do
			if v:IsA("Frame") then
				v:Destroy()
			end
		end
		local OtherIn = game.Players:FindFirstChild(OtherPerson)
		for i, v in pairs(PlayerOrg.Information.Inventory:GetChildren()) do
			local whattt = game.ReplicatedStorage.UrInventorySon:Clone()
			whattt.Name = string.sub(v.Name, 3)
			whattt.Parent = TradeUI.FPL.Inventory
			
			
			
			if Bro[ string.sub(v.Name, 3)] ~= nil then
				whattt.TextButton.UIGradient.Enabled = true
				whattt.TextButton.UIGradient.Color = ColorSequence.new(Bro[ string.sub(v.Name, 3)].Color)
				if Bro[ string.sub(v.Name, 3)].Rotate == true then
					game.TweenService:Create(whattt:FindFirstChild("TextButton"):FindFirstChild("UIGradient"), TweenInfo.new(2.5, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut, -1), {Rotation = 360}):Play()
				else 
					whattt.TextButton.UIGradient.Rotation = 0

				end
			end
			
			
				if string.sub(v.Name, 1, 1) == "K" then
				AddUpToTrade(whattt.TextButton, v, WhichC, Folder)
				whattt.TextButton.Text = string.sub(v.Name, 3)
				elseif string.sub(v.Name, 1, 1) == "R" then
				AddUpToTrade(whattt.TextButton , v, WhichC, Folder)
				whattt.TextButton.Text = string.sub(v.Name, 3)
				elseif string.sub(v.Name, 1, 1) == "T" then
				AddUpToTrade(whattt.TextButton, v, WhichC, Folder)
				
				whattt.TextButton.Text = string.sub(v.Name, 3)
				end
		end

		if WhichC == "B" then
			TradeUI.UI.FP.Text = PlayerOrg.Name
			TradeUI.UI.SP.Text = OtherIn.Name
		else
			TradeUI.UI.FP.Text = OtherIn.Name
			TradeUI.UI.SP.Text = PlayerOrg.Name
		end
		TradeUI.UI.TITLE.Text = "Trade with "..OtherIn.Name
		bener = "Trade with "..OtherIn.Name
		for i, v in pairs(TradeUI.UI:GetDescendants()) do
			if v:IsA("TextButton") and v.Name == "TextButton" then
				v.Text = ""
			end
		end
		AddButton(TradeUI.UI.Accept, Folder, "AcceptTrade")
		AddButton(TradeUI.UI.Decline, Folder, "DeclineTrade")
		TradeUI.FPL.TextLabel.Text = PlayerOrg.Name.."'s inventory"
		for i, v in pairs(OtherIn.Information.Inventory:GetChildren()) do
			local whattt = game.ReplicatedStorage.UrInventorySon:Clone()
			whattt.Parent = TradeUI.SPL.Inventory
			whattt.Name = string.sub(v.Name, 3)
			if string.sub(v.Name, 1, 1) == "K" then
				whattt.TextButton.Text = string.sub(v.Name, 3)
			elseif string.sub(v.Name, 1, 1) == "R" then
				whattt.TextButton.Text = string.sub(v.Name, 3)
			elseif string.sub(v.Name, 1, 1) == "T" then
				whattt.TextButton.Text = string.sub(v.Name, 3)
			end
			
			if Bro[ string.sub(v.Name, 3)] ~= nil then
				whattt.TextButton.UIGradient.Enabled = true
				whattt.TextButton.UIGradient.Color = ColorSequence.new(Bro[ string.sub(v.Name, 3)].Color)
				if Bro[ string.sub(v.Name, 3)].Rotate == true then
					game.TweenService:Create(whattt:FindFirstChild("TextButton"):FindFirstChild("UIGradient"), TweenInfo.new(2.5, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut, -1), {Rotation = 360}):Play()
				else 
					whattt.TextButton.UIGradient.Rotation = 0

				end
			end

			
			
		end
		TradeUI.SPL.TextLabel.Text = OtherIn.Name.."'s inventory"
	end
end
